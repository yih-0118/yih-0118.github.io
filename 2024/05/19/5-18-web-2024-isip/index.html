import requests

def scan_links_in_range(start, end):
    base_url = "http://isip-ctf.tyc4d.tw:8001/user/"
    for i in range(start, end+1):
        link = base_url + str(i)
        response = requests.get(link)
        if response.status_code == 200:
            if response.content:
                print(f"{link}有\n{response.text}")
