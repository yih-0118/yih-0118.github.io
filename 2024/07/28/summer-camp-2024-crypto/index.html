# slove.py
def hex_to_ascii(hex_string):
    hex_list = hex_string.split()
    ascii_string = ''.join([chr(int(hex_num, 16)) for hex_num in hex_list])

def caesar_decrypt(ciphertext, shift):
    plaintext = ""

    return plaintext
